// Testing library tip:
// Quando há necessidade de informar para um biblioteca de testes que um
// elemento está oculto na tela, é possível usar o atributo 'aria-hidden'.
// Exemplo:
<Styled.Wrapper menuVisible={menuVisible} aria-hidden={!menuVisible}>
  {/* O componente acima retorna uma 'nav' */}
  <></>
</Styled.Wrapper>;
